#Events regarding Persia and their relations with India
namespace = abi_persia_extra_events

country_event = {
	id = abi_persia_extra_events.1
	title = abi_persia_extra_events.1.t
	desc = abi_persia_extra_events.1.d
	picture = AFRICAN_INDIAN_eventPicture
	
	fire_only_once = yes
	
	is_triggered_only = yes
	
	option = {	#Accept
		name = abi_persia_extra_events.1.a
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				OR = {
					PER = { religion = zoroastrian }
					PRC = { religion = zoroastrian }
					PRZ = { religion = zoroastrian }
				}
			}
		}
		add_prestige = -25
		custom_tooltip = abi_persia_extra_events.1.tooltip
		if = {
			limit = { exists = PER }
			PER = {
				create_subject = {
					subject_type = satrapy
					subject = ROOT
				}
				add_prestige = 10
				if = {
					limit = { was_tag = SRN }
					ROOT = { override_country_name = INDO_PARTHIA_RENAME }
				}
			}
			hidden_effect = {
				every_owned_province = {
					limit = {
						superregion = persia_superregion
						NOT = { province_id = 2068 }
						NOT = { province_id = 5273 }
					}
					cede_province = PER
					add_core = PER
					remove_core = ROOT
				}
			}
		}
		if = {
			limit = { exists = PRC }
			PRC = {
				create_subject = {
					subject_type = satrapy
					subject = ROOT
				}
				add_prestige = 10
			}
			hidden_effect = {
				every_owned_province = {
					limit = {
						superregion = persia_superregion
						NOT = { province_id = 2068 }
						NOT = { province_id = 5273 }
					}
					cede_province = PRC
					add_core = PRC
					remove_core = ROOT
				}
			}
		}
		if = {
			limit = { exists = PRZ }
			change_religion = sunni
			PRZ = {
				create_subject = {
					subject_type = satrapy
					subject = ROOT
				}
				add_prestige = 10
			}
			hidden_effect = {
				every_owned_province = {
					limit = {
						superregion = persia_superregion
						NOT = { province_id = 2068 }
						NOT = { province_id = 5273 }
					}
					cede_province = PRZ
					add_core = PRZ
					remove_core = ROOT
				}
			}
		}
	}
	
	option = {	#Refuse
		name = abi_persia_extra_events.1.b
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				total_development = 200
			}
			modifier = {
				factor = 2
				total_development = 400
			}
			modifier = {
				factor = 10
				total_development = 600
			}
			modifier = {
				factor = 10
				total_development = PER
			}
			modifier = {
				factor = 10
				total_development = PRC
			}
			modifier = {
				factor = 10
				total_development = PRZ
			}
			modifier = {
				factor = 3
				tag = IDS
			}
		}
		add_war_exhaustion = 2
		if = {
			limit = { exists = PER }
			PER = {
				declare_war_with_cb = {
					who = ROOT
					casus_belli = cb_annex
				}
			}
		}
		if = {
			limit = { exists = PRC }
			PRC = {
				declare_war_with_cb = {
					who = ROOT
					casus_belli = cb_annex
				}
			}
		}
		if = {
			limit = { exists = PRZ }
			PRZ = {
				declare_war_with_cb = {
					who = ROOT
					casus_belli = cb_annex
				}
			}
		}
	}
	
}
